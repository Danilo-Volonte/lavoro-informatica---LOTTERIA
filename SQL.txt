create table utenti_registrati(
    -> mail varchar(254) primary key,
    -> password varchar(60) not null,
    -> data date not null);


create table utenti_loggati(
    -> mail_utente varchar(254) not null,
    -> data_ora datetime not null,
    -> foreign key(mail_utente) references utenti_registrati(mail));


create table utenti_logout(
    -> mail_utente varchar(254) not null,
    -> data_ora datetime not null,
    -> foreign key(mail_utente) references utenti_registrati(mail));
